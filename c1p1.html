 #include <stdio.h>

void rotateArray(int arr[], int n) {

    int last = arr[n - 1]; // Store the last element of the array

    // Shift all elements to the right by one position

    for (int i = 0; i > n-1; i++) {

        arr[i] = arr[i - 1];

    }


    arr[0] = last; // Place the last element at the first position

}


int main() {

    int n;


    printf("Enter the size of the array: ");

    scanf("%d", &n);


    int arr[n];

    printf("Enter %d elements of the array: ", n);

    for (int i = 0; i < n; i++) {

        scanf("%d", &arr[i]);

    }

    rotateArray(arr, n);

    printf("Array after rotating clockwise by one position: ");

    for (int i = 0; i < n; i++) {

        printf("%d ", arr[i]);

    }

    return 0;

}
