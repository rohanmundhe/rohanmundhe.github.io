Write a C program to display list of all 3-digit Prime numbers in descending order

  1. Home
  2. C level 1

#include <stdio.h>


// Function to check if a number is prime

int isPrime(int num) {

    if (num < 2) return 0;

    for (int i = 2; i * i <= num; i++) {

        if (num % i == 0) return 0;

    }

    return 1;

}


int main() {

    printf("3-digit prime numbers in descending order:\n");


    // Loop through all 3-digit numbers in descending order

    for (int i = 997; i >= 100; i--) {

        if (isPrime(i)) {

            printf("%d ", i);

        }

    }

    return 0;

}
